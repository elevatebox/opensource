        // Show/Hide Management Sections
        document.getElementById('coAdminTab').addEventListener('click', function() {
            document.getElementById('coAdminManagement').classList.add('active');
            document.getElementById('studentManagement').classList.remove('active');
            document.getElementById('requestBox').classList.add('active');
        });

        document.getElementById('studentManagementTab').addEventListener('click', function() {
            document.getElementById('studentManagement').classList.add('active');
            document.getElementById('coAdminManagement').classList.remove('active');
            document.getElementById('requestBox').classList.remove('active');
        });

        // Functions to Handle HOD Actions
        function deleteHOD(hodName) {
            alert(`${hodName} has been deleted.`);
            // Logic to remove the HOD from the list (mock)
            const listItem = document.querySelector(`li:contains('${hodName}')`);
            if (listItem) {
                listItem.remove();
                updateHODCount();
            }
        }

        function editHOD(hodName) {
            alert(`Edit HOD: ${hodName}`);
            // Logic for editing HOD (mock)
            const newBranch = prompt("Enter new branch name for " + hodName);
            if (newBranch) {
                const branchSpan = document.querySelector(`li:contains('${hodName}') span`);
                if (branchSpan) {
                    branchSpan.textContent = newBranch;
                }
            }
        }

        function updateHODCount() {
            const hodListItems = document.querySelectorAll('#hodList .list-group-item');
            document.getElementById('totalHODs').textContent = hodListItems.length;
        }
    </script>
